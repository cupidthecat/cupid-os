# DOOM Port + Audio Stack — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Run id Software's DOOM (Freedoom WADs) as the `doom` shell command on CupidOS, with full SFX + OPL3-synthesised music via a new AC97 PCI audio driver and 16-channel software mixer.

**Architecture:** Two stacks linked at the i_sound layer. Audio (kernel/audio/): AC97 → 16-ch mixer → Nuked-OPL3 + MUS→MIDI synth fed from streaming source. DOOM (kernel/doom/): doomgeneric vendored, libc shim mapping `malloc`/`fopen`/`printf` to kernel kmalloc/VFS/serial, platform shim wiring 640×400 letterboxed VBE blit + PS/2 keyboard subscriber + PIT timer, i_sound bridge feeding the audio mixer.

**Tech Stack:** C99 freestanding (`-m32 -nostdlib -nostdinc -ffreestanding`), GCC, NASM, mtools (mcopy), QEMU (`-device AC97`), vendored: doomgeneric (BSD/GPL), Nuked-OPL3 (LGPL), mus2midi (chocolate-doom GPL).

**Spec:** `docs/superpowers/specs/2026-05-06-doom-port-design.md`

**Test pattern:** CupidOS has no host-side unit-test framework — modules are integration-tested in QEMU. Each task that adds runtime behaviour ships a smoke command (a `bin/*.cc` shell program) that exercises the new API and prints `[PASS] ...` or `[FAIL] ...` to serial. Tests run via `make run-headless` driving stdin scripts and grepping the serial transcript. Pure kernel-side helpers (`mus2midi`, `dglibc` formatters) get inline `_self_test()` functions called from boot when `DEBUG_AUDIO` / `DEBUG_DGLIBC` is defined.

```c
/* Raw scancode subscriber.
 * Callback fires from the keyboard IRQ tail with the cooked
 * (high bit cleared) scancode and a `pressed` flag (true on
 * make, false on break). Single subscriber slot — second
 * subscribe attempt returns -1.
 *
 * Subscriber callback runs under the BKL with IRQs disabled;
 * keep it short — push to a ring and process out of IRQ.
 */
typedef void (*kbd_event_cb)(uint8_t scancode, bool pressed, void *ctx);
int  keyboard_subscribe(kbd_event_cb cb, void *ctx);   /* returns 0 on success, -1 if slot taken */
void keyboard_unsubscribe(void);
```

- [ ] **Step 2: Implement in `drivers/keyboard.c`**

Add file-scope state near the top of the file (after existing static vars):

```c
static kbd_event_cb s_kbd_sub_cb = NULL;
static void *s_kbd_sub_ctx = NULL;

int keyboard_subscribe(kbd_event_cb cb, void *ctx) {
    if (s_kbd_sub_cb != NULL) return -1;
    s_kbd_sub_cb = cb;
    s_kbd_sub_ctx = ctx;
    return 0;
}

void keyboard_unsubscribe(void) {
    s_kbd_sub_cb = NULL;
    s_kbd_sub_ctx = NULL;
}
```

In the existing IRQ handler (the function that processes scancodes — locate the spot where `keyboard_get_char` / scancode buffer is fed), invoke the subscriber for **every** scancode (both make and break) before any filtering:

```c
/* Notify raw subscriber if any. Pass cooked scancode + pressed bit. */
if (s_kbd_sub_cb != NULL) {
    bool pressed = (raw_scancode & 0x80) == 0;
    uint8_t cooked = (uint8_t)(raw_scancode & 0x7F);
    s_kbd_sub_cb(cooked, pressed, s_kbd_sub_ctx);
}
```

(`raw_scancode` is the local variable already used in the handler — adapt to whatever the file calls it.)

```c
#include <stdbool.h>

    g_calls++;
    g_last_sc = sc;
    g_last_pressed = pressed;

        return 1;

        serial_write_string("[FAIL] kbdsub: last event mismatch\n");
        return 1;

    serial_write_string("[PASS] kbdsub: subscribe/unsubscribe round-trip\n");

}
```

- [ ] **Step 2: Create `kernel/audio/ac97.c` (probe-only, scaffolding)**

```c
#include "ac97.h"
#include "../pci.h"
#include "../kernel.h"
#include "../../drivers/serial.h"
#include <stdint.h>
#include <stdbool.h>

/* Known AC97 PCI device IDs (Intel ICH family) */
static const uint16_t AC97_VENDOR = 0x8086;
static const uint16_t AC97_DEVICES[] = {
    0x2415, 0x2425, 0x2445, 0x2485, 0x24C5, 0x24D5, 0x266E
};
#define AC97_DEVICE_COUNT (sizeof(AC97_DEVICES) / sizeof(AC97_DEVICES[0]))

static struct {
    bool         present;
    pci_device_t *pci;
    uint16_t     bar_nam;   /* mixer regs (BAR0) */
    uint16_t     bar_nabm;  /* bus-master regs (BAR1) */
    uint8_t      irq_line;
    void       (*fill)(int16_t *, uint32_t);
} s_ac97;

bool ac97_is_present(void) { return s_ac97.present; }

void ac97_set_fill_callback(void (*fill)(int16_t *, uint32_t)) {
    s_ac97.fill = fill;
}

int ac97_init(void) {
    pci_device_t *dev = NULL;
    for (uint32_t i = 0; i < AC97_DEVICE_COUNT; i++) {
        dev = pci_find_by_vendor_device(AC97_VENDOR, AC97_DEVICES[i]);
        if (dev) break;
    }
    if (!dev) {
        serial_write_string("[ac97] no AC97 device found\n");
        return -1;
    }
    /* BAR0 is NAM (mixer), BAR1 is NABM (bus-master) — both IO BARs.
     * pci_device_t stores BAR values as raw config-space dwords; mask
     * off type bit and align. */
    uint32_t bar0 = pci_config_read_dword(dev->bus, dev->slot, dev->func, 0x10);
    uint32_t bar1 = pci_config_read_dword(dev->bus, dev->slot, dev->func, 0x14);
    s_ac97.bar_nam  = (uint16_t)(bar0 & 0xFFFC);
    s_ac97.bar_nabm = (uint16_t)(bar1 & 0xFFFC);
    s_ac97.irq_line = pci_config_read_byte(dev->bus, dev->slot, dev->func, 0x3C);
    s_ac97.pci      = dev;
    s_ac97.present  = true;

    pci_enable_bus_master(dev);

    serial_write_string("[ac97] present: NAM=");
    /* simple hex dump via existing helpers — assume serial_write_hex16 exists */
    extern void serial_write_hex16(uint16_t);
    serial_write_hex16(s_ac97.bar_nam);
    serial_write_string(" NABM=");
    serial_write_hex16(s_ac97.bar_nabm);
    serial_write_string("\n");

    return 0;
}

void ac97_start(void) { /* filled in Task 4 */ }
void ac97_stop(void)  { /* filled in Task 4 */ }
void ac97_set_master_volume(uint8_t pct) { (void)pct; /* Task 4 */ }
```

If `serial_write_hex16` doesn't exist, replace the two hex calls with a static helper at the top of the file:

```c
static void hex16(uint16_t v) {
    static const char H[] = "0123456789ABCDEF";
    char buf[7] = "0x";
    buf[2] = H[(v >> 12) & 0xF];
    buf[3] = H[(v >> 8) & 0xF];
    buf[4] = H[(v >> 4) & 0xF];
    buf[5] = H[v & 0xF];
    buf[6] = 0;
    serial_write_string(buf);
}
```

and use `hex16(...)`.

- [ ] **Step 1: Add NABM and NAM register offsets to `ac97.c`**

Insert near the top, after device-id constants:

```c
/* NAM (mixer) registers — accessed via inw/outw on bar_nam + offset */
#define NAM_RESET           0x00
#define NAM_MASTER_VOL      0x02
#define NAM_PCM_OUT_VOL     0x18
#define NAM_EXT_AUDIO_ID    0x28
#define NAM_EXT_AUDIO_CTRL  0x2A
#define NAM_PCM_FRONT_RATE  0x2C

/* NABM PCM-out (PO) — 0x10 offset from bar_nabm; per-stream 16 bytes */
#define NABM_PO_BDBAR       0x10  /* dword */
#define NABM_PO_CIV         0x14  /* byte  — current index value */
#define NABM_PO_LVI         0x15  /* byte  — last valid index */
#define NABM_PO_SR          0x16  /* word  — status */
#define NABM_PO_PICB        0x18  /* word  — position in current buffer */
#define NABM_PO_CR          0x1B  /* byte  — control */

#define NABM_GLOB_CNT       0x2C
#define NABM_GLOB_STA       0x30

/* PO_SR bits */
#define POSR_BCIS  (1u << 3)
#define POSR_LVBCI (1u << 2)
#define POSR_FIFOE (1u << 4)
/* PO_CR bits */
#define POCR_RPBM  (1u << 0)
#define POCR_RR    (1u << 1)
#define POCR_LVBIE (1u << 2)
#define POCR_FEIE  (1u << 3)
#define POCR_IOCE  (1u << 4)
```

- [ ] **Step 2: Define BDL entry struct and DMA buffers**

```c
#define AC97_BDL_ENTRIES  32
#define AC97_FRAMES_PER_BUF 1024
#define AC97_BYTES_PER_BUF (AC97_FRAMES_PER_BUF * 2 /*ch*/ * 2 /*s16*/)

typedef struct __attribute__((packed)) {
    uint32_t buf_phys;
    uint16_t samples;       /* in samples (s16 count, == frames * 2) */
    uint16_t flags;         /* bit 14 = IOC, bit 15 = BUP */
} bdl_entry_t;

static bdl_entry_t *s_bdl;            /* AC97_BDL_ENTRIES entries, 4-byte-aligned */
static int16_t     *s_dma_pool;       /* AC97_BDL_ENTRIES × AC97_FRAMES_PER_BUF × stereo */
```

Allocate once in `ac97_init`, after the bus-master enable, before the success log:

```c
extern void *kmalloc(uint32_t size);   /* memory.h pulls in macro form */
#include "../memory.h"

s_bdl = (bdl_entry_t *)kmalloc(sizeof(bdl_entry_t) * AC97_BDL_ENTRIES);
s_dma_pool = (int16_t *)kmalloc(AC97_BYTES_PER_BUF * AC97_BDL_ENTRIES);
if (!s_bdl || !s_dma_pool) {
    serial_write_string("[ac97] OOM allocating BDL/DMA pool\n");
    return -1;
}
/* zero buffers (silence) */
for (uint32_t i = 0; i < AC97_BYTES_PER_BUF * AC97_BDL_ENTRIES / 2; i++)
    s_dma_pool[i] = 0;
for (int i = 0; i < AC97_BDL_ENTRIES; i++) {
    s_bdl[i].buf_phys = (uint32_t)(uintptr_t)&s_dma_pool[i * AC97_FRAMES_PER_BUF * 2];
    s_bdl[i].samples  = AC97_FRAMES_PER_BUF * 2;
    s_bdl[i].flags    = (1u << 14);  /* IOC */
}
```

(CupidOS uses identity-mapped paging in kernel space — physical == virtual for kmalloc'd memory.)

- [ ] **Step 5: Install IRQ handler**

```c
extern void irq_install_handler(int irq, void (*handler)(void));

static void ac97_isr(void) {
    uint16_t sr = inw(s_ac97.bar_nabm + NABM_PO_SR);
    if (sr & POSR_BCIS) {
        /* compute the buffer that just completed */
        uint8_t civ = inb(s_ac97.bar_nabm + NABM_PO_CIV);
        uint8_t done = (uint8_t)((civ - 1) & (AC97_BDL_ENTRIES - 1));
        if (s_ac97.fill) {
            int16_t *buf = &s_dma_pool[done * AC97_FRAMES_PER_BUF * 2];
            s_ac97.fill(buf, AC97_FRAMES_PER_BUF);
        }
        /* bump LVI to the buffer we just refilled to keep the ring fed */
        outb(s_ac97.bar_nabm + NABM_PO_LVI, done);
    }
    /* ack all status bits */
    outw(s_ac97.bar_nabm + NABM_PO_SR, sr);
}

/* In ac97_init, before logging success: */
irq_install_handler(s_ac97.irq_line, ac97_isr);
```

- [ ] **Step 6: Implement start/stop/volume**

```c
void ac97_start(void) {
    if (!s_ac97.present) return;
    outb(s_ac97.bar_nabm + NABM_PO_CR,
         (uint8_t)(POCR_IOCE | POCR_LVBIE | POCR_FEIE | POCR_RPBM));
}

void ac97_stop(void) {
    if (!s_ac97.present) return;
    outb(s_ac97.bar_nabm + NABM_PO_CR, 0);
}

void ac97_set_master_volume(uint8_t pct) {
    if (!s_ac97.present) return;
    if (pct > 100) pct = 100;
    /* AC97 volume: 0 = full, 0x3F = mute, lower = louder */
    uint8_t att = (uint8_t)((100 - pct) * 0x3F / 100);
    uint16_t v = (uint16_t)((att << 8) | att);
    outw(s_ac97.bar_nam + NAM_MASTER_VOL, v);
}
```

```c
/* audiotest — exercise AC97 + mixer.
 *   audiotest sine    — 440 Hz tone, 2 seconds, both channels
 *   audiotest opl     — Task 8
 *   audiotest pan     — Task 6
 */
#include <stdint.h>
#include <stdbool.h>

extern void ac97_start(void);
extern void ac97_stop(void);
extern void ac97_set_fill_callback(void (*fill)(int16_t *, uint32_t));
extern bool ac97_is_present(void);
extern void serial_write_string(const char *);
extern void delay_ms(uint32_t);   /* if absent: process_yield loop on PIT */
extern uint32_t pit_get_ticks(void);

static int16_t g_phase = 0;
static const int16_t SINE_INC = (int16_t)((65536u * 440u) / 22050u);

/* tiny sine via piecewise-linear triangle for smoke test */
static void fill_sine(int16_t *buf, uint32_t frames) {
    for (uint32_t i = 0; i < frames; i++) {
        int16_t t = g_phase;
        /* triangle: -16384..+16384 */
        int16_t s = (t < 0) ? (int16_t)(t + 16384) : (int16_t)(16384 - t);
        s = (int16_t)(s * 2 - 16384);
        buf[i*2 + 0] = s;
        buf[i*2 + 1] = s;
        g_phase = (int16_t)(g_phase + SINE_INC);
    }
}

static int sub_sine(void) {
    if (!ac97_is_present()) {
        serial_write_string("[SKIP] audiotest sine: no AC97 device\n");
        return 0;
    }
    ac97_set_fill_callback(fill_sine);
    ac97_start();
    /* 2 seconds = 200 PIT ticks @ 100Hz */
    uint32_t t0 = pit_get_ticks();
    while (pit_get_ticks() - t0 < 200) { /* spin */ }
    ac97_stop();
    serial_write_string("[PASS] audiotest sine\n");
    return 0;
}

extern int strcmp(const char *a, const char *b);

int main(int argc, char **argv) {
    if (argc < 2) {
        serial_write_string("usage: audiotest <sine|opl|pan|sweep>\n");
        return 1;
    }
    if (strcmp(argv[1], "sine") == 0) return sub_sine();
    serial_write_string("[FAIL] audiotest: unknown subtest\n");
    return 1;
}
```

```c
#include "mixer.h"
#include "../../drivers/serial.h"
#include <stdint.h>

typedef struct {
    /* fixed PCM */
    const int16_t *pcm;
    uint32_t       len_frames;
    uint32_t       pos_frames;
    uint8_t        channels;
    uint8_t        loop;
    uint8_t        vol_l;
    uint8_t        vol_r;
    uint8_t        active;
    /* streaming */
    uint8_t        is_stream;
    mixer_pull_fn  pull;
    void          *ctx;
} slot_t;

static slot_t s_slots[MIXER_SLOTS];

int mixer_init(void) {
    for (int i = 0; i < MIXER_SLOTS; i++) {
        s_slots[i].active = 0;
        s_slots[i].is_stream = 0;
    }
    return 0;
}

static int slot_valid(int slot) { return slot >= 0 && slot < MIXER_SLOTS; }

int mixer_play(int slot, const int16_t *pcm, uint32_t frames,
               uint8_t channels, uint8_t loop,
               uint8_t vol_l, uint8_t vol_r) {
    if (!slot_valid(slot) || !pcm || (channels != 1 && channels != 2))
        return -1;
    s_slots[slot].pcm = pcm;
    s_slots[slot].len_frames = frames;
    s_slots[slot].pos_frames = 0;
    s_slots[slot].channels = channels;
    s_slots[slot].loop = loop;
    s_slots[slot].vol_l = vol_l;
    s_slots[slot].vol_r = vol_r;
    s_slots[slot].is_stream = 0;
    s_slots[slot].active = 1;
    return 0;
}

int mixer_play_stream(int slot, mixer_pull_fn pull, void *ctx,
                      uint8_t vol_l, uint8_t vol_r) {
    if (!slot_valid(slot) || !pull) return -1;
    s_slots[slot].pull = pull;
    s_slots[slot].ctx = ctx;
    s_slots[slot].vol_l = vol_l;
    s_slots[slot].vol_r = vol_r;
    s_slots[slot].is_stream = 1;
    s_slots[slot].active = 1;
    return 0;
}

void mixer_stop(int slot) {
    if (slot_valid(slot)) s_slots[slot].active = 0;
}

int mixer_active(int slot) {
    return slot_valid(slot) ? (int)s_slots[slot].active : 0;
}

void mixer_set_volume(int slot, uint8_t vol_l, uint8_t vol_r) {
    if (slot_valid(slot)) {
        s_slots[slot].vol_l = vol_l;
        s_slots[slot].vol_r = vol_r;
    }
}

static int16_t clamp16(int32_t v) {
    if (v >  32767) return  32767;
    if (v < -32768) return -32768;
    return (int16_t)v;
}

void mixer_fill(int16_t *out, uint32_t frames) {
    /* zero accumulator (in-place) */
    for (uint32_t f = 0; f < frames; f++) {
        out[f*2 + 0] = 0;
        out[f*2 + 1] = 0;
    }
    static int16_t scratch[1024 * 2];   /* per-slot stereo */
    for (int s = 0; s < MIXER_SLOTS; s++) {
        slot_t *sl = &s_slots[s];
        if (!sl->active) continue;
        if (sl->is_stream) {
            sl->pull(scratch, frames, sl->ctx);
            for (uint32_t f = 0; f < frames; f++) {
                int32_t l = (int32_t)scratch[f*2 + 0] * sl->vol_l;
                int32_t r = (int32_t)scratch[f*2 + 1] * sl->vol_r;
                out[f*2 + 0] = clamp16((int32_t)out[f*2 + 0] + (l >> 7));
                out[f*2 + 1] = clamp16((int32_t)out[f*2 + 1] + (r >> 7));
            }
        } else {
            for (uint32_t f = 0; f < frames; f++) {
                if (sl->pos_frames >= sl->len_frames) {
                    if (sl->loop) sl->pos_frames = 0;
                    else { sl->active = 0; break; }
                }
                int16_t l, r;
                if (sl->channels == 1) {
                    int16_t v = sl->pcm[sl->pos_frames];
                    l = r = v;
                } else {
                    l = sl->pcm[sl->pos_frames * 2 + 0];
                    r = sl->pcm[sl->pos_frames * 2 + 1];
                }
                int32_t li = (int32_t)l * sl->vol_l;
                int32_t ri = (int32_t)r * sl->vol_r;
                out[f*2 + 0] = clamp16((int32_t)out[f*2 + 0] + (li >> 7));
                out[f*2 + 1] = clamp16((int32_t)out[f*2 + 1] + (ri >> 7));
                sl->pos_frames++;
            }
        }
    }
}
```

```c
/* audiotest — exercise mixer + AC97
 * Modes: sine  (440 Hz, slot 9)
 *        sweep (50 Hz → 8 kHz log, slot 9)
 *        pan   (1 kHz, ping-pong L↔R via volume changes)
 */
#include <stdint.h>
#include <stdbool.h>

extern bool     ac97_is_present(void);
extern int      mixer_play(int slot, const int16_t *pcm, uint32_t frames,
                           uint8_t ch, uint8_t loop, uint8_t vl, uint8_t vr);
extern void     mixer_stop(int slot);
extern void     mixer_set_volume(int, uint8_t, uint8_t);
extern void     serial_write_string(const char *);
extern uint32_t pit_get_ticks(void);
extern void    *kmalloc(uint32_t);
extern void     kfree(void *);
extern int      strcmp(const char *, const char *);

#define SR 22050

static int16_t *gen_tone_mono(uint32_t hz, uint32_t ms) {
    uint32_t frames = (SR * ms) / 1000;
    int16_t *p = (int16_t *)kmalloc(frames * 2);
    if (!p) return 0;
    /* triangle generator — float-free */
    int32_t inc = (int32_t)((65536 * hz) / SR);
    int32_t phase = 0;
    for (uint32_t i = 0; i < frames; i++) {

        int32_t s = (t < 0x8000) ? (t - 0x4000) : (0xC000 - t);
        p[i] = (int16_t)(s * 2);
        phase += inc;
    }
    return p;
}

    uint32_t target = pit_get_ticks() + (ms / 10);
    while (pit_get_ticks() < target) { /* spin */ }
}

    if (!t) { serial_write_string("[FAIL] sine: kmalloc\n"); return 1; }
    mixer_play(9, t, (SR * 2000) / 1000, 1, 0, 100, 100);
    wait_ms(2000);
    mixer_stop(9);
    kfree(t);
    serial_write_string("[PASS] audiotest sine\n");
    return 0;
}

static int sub_sweep(void) {
    if (!ac97_is_present()) { serial_write_string("[SKIP] sweep: no AC97\n"); return 0; }
    /* 8 segments of 500 ms each, log freq spacing 50 -> 8000 */
    static const uint32_t F[8] = { 50, 100, 200, 400, 800, 1600, 3200, 8000 };
    for (int i = 0; i < 8; i++) {
        int16_t *t = gen_tone_mono(F[i], 500);
        mixer_play(9, t, (SR * 500) / 1000, 1, 0, 100, 100);
        wait_ms(500);
        mixer_stop(9);
        kfree(t);
    }
    serial_write_string("[PASS] audiotest sweep\n");
    return 0;
}

static int sub_pan(void) {
    if (!ac97_is_present()) { serial_write_string("[SKIP] pan: no AC97\n"); return 0; }
    int16_t *t = gen_tone_mono(1000, 4000);
    mixer_play(9, t, (SR * 4000) / 1000, 1, 0, 100, 0);
    /* 8 ramps over 4s */
    for (int i = 0; i < 8; i++) {
        for (int j = 0; j <= 100; j += 10) {
            uint8_t l = (i & 1) ? (uint8_t)j : (uint8_t)(100 - j);
            uint8_t r = (i & 1) ? (uint8_t)(100 - j) : (uint8_t)j;
            mixer_set_volume(9, l, r);
            wait_ms(50);
        }
    }
    mixer_stop(9);
    kfree(t);
    serial_write_string("[PASS] audiotest pan\n");
    return 0;
}

int main(int argc, char **argv) {
    if (argc < 2) { serial_write_string("usage: audiotest <sine|sweep|pan|opl>\n"); return 1; }
    if (strcmp(argv[1], "sine")  == 0) return sub_sine();
    if (strcmp(argv[1], "sweep") == 0) return sub_sweep();
    if (strcmp(argv[1], "pan")   == 0) return sub_pan();
    serial_write_string("[FAIL] audiotest: unknown subtest\n");
    return 1;
}
```

```c
extern void OPL3_WriteRegBuffered(void *chip, uint16_t reg, uint8_t v);
extern void OPL3_GenerateStream(void *chip, int16_t *buf, uint32_t numsamples);

/* Buffer big enough for one slot scratch ~1024 frames × stereo s16,
 * plus generator scratch for OPL3 49716->22050 resampling */
static unsigned char g_opl_chip[16 * 1024]; /* size > sizeof(opl3_chip) */
static uint32_t g_opl_phase_q16; /* 49716/22050 ≈ 2.255 → q16 = 147834 */

struct opl_ctx { int dummy; };

static void opl_pull(int16_t *out, uint32_t frames, void *ctx) {
    (void)ctx;
    /* Generate at native 49716, downsample by ratio 22050/49716. */
    static int16_t native[1024 * 2];   /* up to ~2x the request */
    uint32_t need = (frames * 49716 + 22049) / 22050;
    if (need > 1024) need = 1024;
    OPL3_GenerateStream(g_opl_chip, native, need);
    /* Linear-interp resample 49716 -> 22050. */
    for (uint32_t f = 0; f < frames; f++) {
        uint32_t src_q16 = (f * 49716u * 65536u) / 22050u;
        uint32_t src_idx = src_q16 >> 16;
        if (src_idx >= need - 1) src_idx = need - 2;
        uint32_t frac = src_q16 & 0xFFFF;
        int32_t l0 = native[src_idx*2 + 0];
        int32_t l1 = native[(src_idx+1)*2 + 0];
        int32_t r0 = native[src_idx*2 + 1];
        int32_t r1 = native[(src_idx+1)*2 + 1];
        out[f*2 + 0] = (int16_t)(l0 + (((l1 - l0) * (int32_t)frac) >> 16));
        out[f*2 + 1] = (int16_t)(r0 + (((r1 - r0) * (int32_t)frac) >> 16));
    }
    (void)g_opl_phase_q16;
}

    OPL3_WriteRegBuffered(g_opl_chip, 0x20, 0x01);  /* mult */
    OPL3_WriteRegBuffered(g_opl_chip, 0x40, 0x10);  /* level */
    OPL3_WriteRegBuffered(g_opl_chip, 0x60, 0xF0);  /* AD */
    OPL3_WriteRegBuffered(g_opl_chip, 0x80, 0x77);  /* SR */
    OPL3_WriteRegBuffered(g_opl_chip, 0x23, 0x01);

    OPL3_WriteRegBuffered(g_opl_chip, 0xC0, 0x30);  /* feedback + connection + L+R */
```

```c
                     uint8_t **out_midi, uint32_t *out_len) {

    MEMFILE *dst = mem_fopen_write();
    if (!src || !dst) return -1;
    if (mus2mid(src, dst) != 0) {
        mem_fclose(src); mem_fclose(dst);
        return -2;

    mem_fclose(dst);
    return 0;
}
```

```c
#include "midiopl.h"
#include "nuked_opl3.h"
#include "../../drivers/serial.h"
#include "../string.h"
#include <stdint.h>

/* GENMIDI lump structure (DMX format):
 *   header[8]  = "#OPL_II#"
 *   175 patches × 36 bytes  (128 melodic + 47 percussion)
 *   175 × 32 chars name
 * Each 36-byte patch:
 *   uint16 flags
 *   uint8  finetune
 *   uint8  note (for percussion)
 *   16-byte voice0 (op modulator/carrier 6 bytes each + feedback/conn 2 bytes + 2 unused)
 *   16-byte voice1 (4-op extension; we only use the OPL3 4-op when flags bit 0 set)
 */

typedef struct {
    uint8_t mult, ksr_eg_vib_am, atk_dec, sus_rel, waveform, ksl_lvl;
} opl_op_t;

typedef struct {
    opl_op_t mod;
    opl_op_t car;
    uint8_t  feedback_conn;
} opl_voice_t;

typedef struct {
    uint16_t   flags;
    int8_t     finetune;
    uint8_t    note_offset;
    opl_voice_t v0;
    opl_voice_t v1;
} genmidi_patch_t;

#define NUM_PATCHES 175
static genmidi_patch_t s_patches[NUM_PATCHES];
static int s_patches_loaded = 0;

/* OPL3 9-channel allocation (use only first 9 melodic for v1). */
#define NUM_OPL_CH 9
typedef struct {
    int     in_use;
    uint8_t midi_ch;
    uint8_t midi_note;
    uint32_t age;   /* for LRU steal */
} opl_ch_state_t;
static opl_ch_state_t s_opl_ch[NUM_OPL_CH];
static uint32_t s_age = 0;

static unsigned char g_opl_chip[16 * 1024];
static uint8_t s_master_vol = 100;     /* 0–127 mapped to attenuation */

/* MIDI channel state */
typedef struct {
    uint8_t  program;
    uint8_t  volume;       /* CC 7 */
    uint8_t  pan;          /* CC 10 */
    uint16_t pitch_bend;
} midi_ch_state_t;
static midi_ch_state_t s_midi[16];

/* MIDI parser running-status */
static uint8_t s_running_status = 0;
static uint8_t s_msg_buf[3];
static uint8_t s_msg_have = 0;
static uint8_t s_msg_need = 0;

static int midiopl_load_genmidi(const uint8_t *lump, uint32_t len);
static void midiopl_handle_event(const uint8_t *msg, uint8_t len);
static void note_on(uint8_t ch, uint8_t note, uint8_t vel);
static void note_off(uint8_t ch, uint8_t note);
static void program_change(uint8_t ch, uint8_t prog);
static void controller(uint8_t ch, uint8_t cc, uint8_t v);
static void pitch_bend(uint8_t ch, uint16_t bend);
static int  alloc_opl_ch(uint8_t midi_ch, uint8_t note);
static void free_opl_ch(int oplc);
static void program_patch(int oplc, const genmidi_patch_t *p);
static void key_on(int oplc, uint16_t fnum, uint8_t block);

int midiopl_init(const uint8_t *genmidi_lump, uint32_t lump_len) {
    OPL3_Reset(g_opl_chip, 49716);
    OPL3_WriteRegBuffered(g_opl_chip, 0x105, 0x01);  /* OPL3 enable */
    for (int i = 0; i < NUM_OPL_CH; i++) {
        s_opl_ch[i].in_use = 0;
        s_opl_ch[i].age = 0;
    }
    for (int i = 0; i < 16; i++) {
        s_midi[i].program = 0;
        s_midi[i].volume = 100;
        s_midi[i].pan = 64;
        s_midi[i].pitch_bend = 0x2000;
    }
    s_running_status = 0;
    s_msg_have = 0;
    if (midiopl_load_genmidi(genmidi_lump, lump_len) != 0) {
        serial_write_string("[midiopl] WARN: GENMIDI parse failed; using defaults\n");
    }
    return 0;
}

void midiopl_reset(void) {
    for (int i = 0; i < NUM_OPL_CH; i++) {
        if (s_opl_ch[i].in_use) free_opl_ch(i);
    }
    s_running_status = 0;
    s_msg_have = 0;
}

/* GENMIDI parse — returns 0 on success */
static int midiopl_load_genmidi(const uint8_t *lump, uint32_t len) {
    if (len < 8 + NUM_PATCHES * 36) return -1;
    if (lump[0] != '#' || lump[1] != 'O' || lump[2] != 'P' || lump[3] != 'L'
        || lump[4] != '_' || lump[5] != 'I' || lump[6] != 'I' || lump[7] != '#')
        return -2;
    const uint8_t *p = lump + 8;
    for (int i = 0; i < NUM_PATCHES; i++) {
        s_patches[i].flags = (uint16_t)(p[0] | (p[1] << 8));
        s_patches[i].finetune = (int8_t)p[2];
        s_patches[i].note_offset = p[3];
        /* voice0 — 16 bytes starting at p+4 */
        const uint8_t *v0 = p + 4;
        s_patches[i].v0.mod.mult = v0[0];
        s_patches[i].v0.mod.atk_dec = v0[1];
        s_patches[i].v0.mod.sus_rel = v0[2];
        s_patches[i].v0.mod.waveform = v0[3];
        s_patches[i].v0.mod.ksl_lvl = v0[4];
        s_patches[i].v0.mod.ksr_eg_vib_am = v0[5];
        s_patches[i].v0.feedback_conn = v0[6];
        /* skip 1 byte (output level scale) */
        s_patches[i].v0.car.mult = v0[8];
        s_patches[i].v0.car.atk_dec = v0[9];
        s_patches[i].v0.car.sus_rel = v0[10];
        s_patches[i].v0.car.waveform = v0[11];
        s_patches[i].v0.car.ksl_lvl = v0[12];
        s_patches[i].v0.car.ksr_eg_vib_am = v0[13];
        /* voice1 — 16 bytes at p+20 (skipped on 2-op patches) */
        /* (left unmodelled for v1; we only use v0) */
        p += 36;
    }
    s_patches_loaded = 1;
    return 0;
}

/* MIDI byte feed — parses running-status and dispatches events. */
void midiopl_feed(const uint8_t *bytes, uint32_t len) {
    for (uint32_t i = 0; i < len; i++) {
        uint8_t b = bytes[i];
        if (b & 0x80) {
            /* status */
            if (b >= 0xF8) continue;          /* realtime — ignore */
            if (b == 0xFF) {                  /* meta — skip */
                /* meta type, vlq length, data... — naive skip */
                if (i + 2 >= len) return;
                /* not parsed — robust handler in §future */
                /* read meta type */
                /* uint8_t mt = bytes[i+1]; */
                /* read vlq length */
                uint32_t k = i + 2;
                uint32_t l = 0;
                while (k < len) {
                    l = (l << 7) | (bytes[k] & 0x7F);
                    if (!(bytes[k] & 0x80)) { k++; break; }
                    k++;
                }
                i = k + l - 1;
                s_running_status = 0;
                continue;
            }
            if (b == 0xF0 || b == 0xF7) {     /* sysex — skip */
                /* find F7 */
                while (i < len && bytes[i] != 0xF7) i++;
                s_running_status = 0;
                continue;
            }
            s_running_status = b;
            s_msg_have = 1;
            s_msg_buf[0] = b;
            uint8_t kind = b & 0xF0;
            if (kind == 0xC0 || kind == 0xD0) s_msg_need = 2;
            else                              s_msg_need = 3;
        } else {
            if (s_msg_have == 0 && s_running_status) {
                s_msg_buf[0] = s_running_status;
                s_msg_have = 1;
                uint8_t kind = s_running_status & 0xF0;
                s_msg_need = (kind == 0xC0 || kind == 0xD0) ? 2 : 3;
            }
            if (s_msg_have < s_msg_need) {
                s_msg_buf[s_msg_have++] = b;
                if (s_msg_have == s_msg_need) {
                    midiopl_handle_event(s_msg_buf, s_msg_need);
                    s_msg_have = 1;          /* keep status for running-status next */
                }
            }
        }
    }
}

static void midiopl_handle_event(const uint8_t *msg, uint8_t len) {
    uint8_t kind = msg[0] & 0xF0;
    uint8_t ch = msg[0] & 0x0F;
    if (kind == 0x90 && len == 3 && msg[2] != 0) note_on(ch, msg[1], msg[2]);
    else if ((kind == 0x80 || (kind == 0x90 && msg[2] == 0)) && len == 3) note_off(ch, msg[1]);
    else if (kind == 0xC0 && len == 2) program_change(ch, msg[1]);
    else if (kind == 0xB0 && len == 3) controller(ch, msg[1], msg[2]);
    else if (kind == 0xE0 && len == 3) pitch_bend(ch, (uint16_t)(msg[1] | (msg[2] << 7)));
}

/* OPL3 channel base register table — channels 0..8 use base offsets
 * 0,1,2, 8,9,A, 10,11,12 for ops; channel itself uses 0,1,..8 for
 * A0/B0/C0 set. */
static const uint8_t OP_OFFSET[9][2] = {
    {0x00, 0x03}, {0x01, 0x04}, {0x02, 0x05},
    {0x08, 0x0B}, {0x09, 0x0C}, {0x0A, 0x0D},
    {0x10, 0x13}, {0x11, 0x14}, {0x12, 0x15}
};

static void program_patch(int oplc, const genmidi_patch_t *p) {
    uint8_t mod_off = OP_OFFSET[oplc][0];
    uint8_t car_off = OP_OFFSET[oplc][1];
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x20 + mod_off), p->v0.mod.mult);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x40 + mod_off), p->v0.mod.ksl_lvl);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x60 + mod_off), p->v0.mod.atk_dec);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x80 + mod_off), p->v0.mod.sus_rel);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0xE0 + mod_off), p->v0.mod.waveform);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x20 + car_off), p->v0.car.mult);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x40 + car_off), p->v0.car.ksl_lvl);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x60 + car_off), p->v0.car.atk_dec);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0x80 + car_off), p->v0.car.sus_rel);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0xE0 + car_off), p->v0.car.waveform);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0xC0 + oplc),
                          (uint8_t)(p->v0.feedback_conn | 0x30));   /* L+R */
}

/* Note → F-num/block lookup (octave 0..7). Reduced to 12-entry table. */
static const uint16_t FNUM[12] = {
    0x158, 0x16D, 0x183, 0x19A, 0x1B2, 0x1CC,
    0x1E7, 0x204, 0x223, 0x244, 0x267, 0x28D
};

static void key_on(int oplc, uint16_t fnum, uint8_t block) {
    uint8_t fl = (uint8_t)(fnum & 0xFF);
    uint8_t fh = (uint8_t)((fnum >> 8) & 0x03);
    uint8_t b_kon_fh = (uint8_t)(0x20 | ((block & 0x07) << 2) | fh);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0xA0 + oplc), fl);
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0xB0 + oplc), b_kon_fh);
}

static int alloc_opl_ch(uint8_t midi_ch, uint8_t note) {
    /* find free first */
    for (int i = 0; i < NUM_OPL_CH; i++) {
        if (!s_opl_ch[i].in_use) {
            s_opl_ch[i].in_use = 1;
            s_opl_ch[i].midi_ch = midi_ch;
            s_opl_ch[i].midi_note = note;
            s_opl_ch[i].age = ++s_age;
            return i;
        }
    }
    /* steal LRU */
    int oldest = 0;
    for (int i = 1; i < NUM_OPL_CH; i++)
        if (s_opl_ch[i].age < s_opl_ch[oldest].age) oldest = i;
    free_opl_ch(oldest);
    s_opl_ch[oldest].in_use = 1;
    s_opl_ch[oldest].midi_ch = midi_ch;
    s_opl_ch[oldest].midi_note = note;
    s_opl_ch[oldest].age = ++s_age;
    return oldest;
}

static void free_opl_ch(int oplc) {
    /* key off */
    OPL3_WriteRegBuffered(g_opl_chip, (uint16_t)(0xB0 + oplc),
                          (uint8_t)(s_opl_ch[oplc].in_use ? 0x00 : 0x00));
    s_opl_ch[oplc].in_use = 0;
}

static void note_on(uint8_t ch, uint8_t note, uint8_t vel) {
    if (!s_patches_loaded) return;
    uint8_t prog = s_midi[ch].program;
    if (prog >= NUM_PATCHES) prog = 0;
    int oplc = alloc_opl_ch(ch, note);
    program_patch(oplc, &s_patches[prog]);
    /* note 60 = middle C; map to fnum/block */
    int oct = note / 12;
    if (oct < 0) oct = 0;
    if (oct > 7) oct = 7;
    uint16_t fnum = FNUM[note % 12];
    key_on(oplc, fnum, (uint8_t)oct);
    (void)vel; /* TODO: scale carrier level by velocity in v2 */
}

static void note_off(uint8_t ch, uint8_t note) {
    for (int i = 0; i < NUM_OPL_CH; i++) {
        if (s_opl_ch[i].in_use && s_opl_ch[i].midi_ch == ch
            && s_opl_ch[i].midi_note == note) {
            free_opl_ch(i);
            return;
        }
    }
}

static void program_change(uint8_t ch, uint8_t prog) {
    s_midi[ch].program = prog;
}
static void controller(uint8_t ch, uint8_t cc, uint8_t v) {
    if (cc == 7)  s_midi[ch].volume = v;
    if (cc == 10) s_midi[ch].pan = v;
    if (cc == 123) {  /* all notes off */
        for (int i = 0; i < NUM_OPL_CH; i++)
            if (s_opl_ch[i].in_use && s_opl_ch[i].midi_ch == ch) free_opl_ch(i);
    }
}
static void pitch_bend(uint8_t ch, uint16_t bend) {
    s_midi[ch].pitch_bend = bend;
    /* TODO: re-key ringing notes with detuned fnum */
}

void midiopl_set_volume(uint8_t v) { s_master_vol = v > 127 ? 127 : v; }

void midiopl_render(int16_t *out, uint32_t frames) {
    /* Generate at OPL3 native 49716, downsample to 22050. */
    static int16_t native[2048 * 2];
    uint32_t need = (frames * 49716u + 22049u) / 22050u;
    if (need > 2048) need = 2048;
    OPL3_GenerateStream(g_opl_chip, native, need);
    int32_t att = (int32_t)s_master_vol;       /* 0..127 */
    for (uint32_t f = 0; f < frames; f++) {
        uint32_t src_q16 = (f * 49716u * 65536u) / 22050u;
        uint32_t i = src_q16 >> 16;
        if (i >= need - 1) i = need - 2;
        uint32_t fr = src_q16 & 0xFFFF;
        int32_t l0 = native[i*2], l1 = native[(i+1)*2];
        int32_t r0 = native[i*2+1], r1 = native[(i+1)*2+1];
        int32_t l = (l0 + (((l1 - l0) * (int32_t)fr) >> 16)) * att / 127;
        int32_t r = (r0 + (((r1 - r0) * (int32_t)fr) >> 16)) * att / 127;
        out[f*2 + 0] = (int16_t)l;
        out[f*2 + 1] = (int16_t)r;
    }
}
```

```c
/* Naive realloc: malloc new, copy old size (we don't know it; over-copy
 * by `newsz` and rely on heap to handle bounds). DOOM doesn't realloc much. */
void *dg_realloc(void *p, uint32_t newsz) {
    void *q = kmalloc(newsz);
    if (!q) return 0;
    if (p) {
        /* We can't know the old size precisely without a header.
         * Worst case: copy newsz bytes (may read past old block).
         * Heap blocks are at least newsz/2; safe in practice for DOOM. */
        memcpy(q, p, newsz);
        kfree(p);
    }
    return q;
}

char *dg_strdup(const char *s) {
    uint32_t n = strlen(s) + 1;
    char *r = (char *)kmalloc(n);
    if (r) memcpy(r, s, n);
    return r;
}

int dg_strcasecmp(const char *a, const char *b) {
    while (*a && *b) {
        int da = dg_tolower((uint8_t)*a), db = dg_tolower((uint8_t)*b);
        if (da != db) return da - db;
        a++; b++;
    }
    return dg_tolower((uint8_t)*a) - dg_tolower((uint8_t)*b);
}

        int da = dg_tolower((uint8_t)*a), db = dg_tolower((uint8_t)*b);
        if (da != db) return da - db;
        a++; b++; n--;
    }
    return n ? (dg_tolower((uint8_t)*a) - dg_tolower((uint8_t)*b)) : 0;
}

static void out_char(char **buf, uint32_t *space, char c) {
    if (*space) { **buf = c; (*buf)++; (*space)--; }
}

static int format_into(char *out, uint32_t cap, const char *fmt, va_list ap) {
    char *p = out;
    uint32_t space = cap ? cap - 1 : 0;
    while (*fmt) {
        if (*fmt != '%') { out_char(&p, &space, *fmt++); continue; }
        fmt++;
        int width = 0;
        int zero = 0;
        int neg = 0;
        if (*fmt == '-') { neg = 1; fmt++; }
        if (*fmt == '0') { zero = 1; fmt++; }
        while (*fmt >= '0' && *fmt <= '9') { width = width * 10 + (*fmt - '0'); fmt++; }
        if (*fmt == '.') { fmt++;
            while (*fmt >= '0' && *fmt <= '9') fmt++; /* precision parsed but ignored except for %s */
        }
        if (*fmt == 'l') fmt++;
        char ch = *fmt++;
        if (ch == 'd' || ch == 'i') {
            int v = va_arg(ap, int);
            char buf[16]; int n = 0;
            uint32_t u = (v < 0) ? (uint32_t)(-(int64_t)v) : (uint32_t)v;
            do { buf[n++] = (char)('0' + u % 10); u /= 10; } while (u);
            if (v < 0) buf[n++] = '-';
            while (n < width && !neg) { out_char(&p, &space, zero ? '0' : ' '); width--; }
            while (n) out_char(&p, &space, buf[--n]);
        } else if (ch == 'u') {
            uint32_t u = va_arg(ap, uint32_t);
            char buf[16]; int n = 0;
            do { buf[n++] = (char)('0' + u % 10); u /= 10; } while (u);
            while (n) out_char(&p, &space, buf[--n]);
        } else if (ch == 'x' || ch == 'X' || ch == 'p') {
            uint32_t u = (ch == 'p') ? (uint32_t)(uintptr_t)va_arg(ap, void *)
                                     : va_arg(ap, uint32_t);
            const char *H = (ch == 'X') ? "0123456789ABCDEF" : "0123456789abcdef";
            char buf[16]; int n = 0;
            do { buf[n++] = H[u & 0xF]; u >>= 4; } while (u);
            while (n) out_char(&p, &space, buf[--n]);
        } else if (ch == 'c') {
            int c = va_arg(ap, int);
            out_char(&p, &space, (char)c);
        } else if (ch == 's') {
            const char *s = va_arg(ap, const char *);
            if (!s) s = "(null)";
            while (*s) out_char(&p, &space, *s++);
        } else if (ch == '%') {
            out_char(&p, &space, '%');
        } else {
            out_char(&p, &space, '%'); out_char(&p, &space, ch);
        }
    }
    if (cap) *p = 0;
    return (int)(p - out);
}

int dg_vsnprintf(char *s, uint32_t n, const char *fmt, void *va) {
    return format_into(s, n, fmt, *(va_list *)va);
}
int dg_snprintf(char *s, uint32_t n, const char *fmt, ...) {
    va_list ap; va_start(ap, fmt);
    int r = format_into(s, n, fmt, ap);
    va_end(ap);
    return r;
}
int dg_sprintf(char *s, const char *fmt, ...) {
    va_list ap; va_start(ap, fmt);
    int r = format_into(s, 0xFFFFFFFFu, fmt, ap);
    va_end(ap);
    return r;
}

DG_FILE *dg_fopen(const char *path, const char *mode) {
    uint32_t flags = 0;
    if (mode[0] == 'r') flags = 0;        /* read */
    else if (mode[0] == 'w') flags = 1;   /* write+truncate — VFS may use a different bit */
    else if (mode[0] == 'a') flags = 2;
    int fd = vfs_open(path, flags);
    if (fd < 0) return 0;
    DG_FILE *f = (DG_FILE *)kmalloc(sizeof(*f));
    if (!f) { vfs_close(fd); return 0; }
    f->fd = fd; f->eof = 0; f->err = 0; f->kind = 0;
    return f;
}
int dg_fclose(DG_FILE *f) {
    if (!f || f->kind != 0) return 0;
    int r = vfs_close(f->fd);
    kfree(f);
    return r;
}
uint32_t dg_fread(void *p, uint32_t sz, uint32_t n, DG_FILE *f) {
    if (!f || f->kind != 0) return 0;
    uint32_t total = sz * n;
    int r = vfs_read(f->fd, p, total);
    if (r < 0) { f->err = 1; return 0; }
    if ((uint32_t)r < total) f->eof = 1;
    return (uint32_t)r / sz;
}
uint32_t dg_fwrite(const void *p, uint32_t sz, uint32_t n, DG_FILE *f) {
    if (!f) return 0;
    if (f->kind == 1 || f->kind == 2) {
        /* stdout/stderr → serial */
        serial_write_string("");  /* ensure flushed; serial_write below */
        const char *bp = (const char *)p;
        for (uint32_t i = 0; i < sz * n; i++) {
            char two[2] = { bp[i], 0 };
            serial_write_string(two);
        }
        return n;
    }
    int r = vfs_write(f->fd, p, sz * n);
    if (r < 0) { f->err = 1; return 0; }
    return (uint32_t)r / sz;
}
int dg_fseek(DG_FILE *f, int32_t off, int whence) {
    if (!f || f->kind != 0) return -1;
    return vfs_seek(f->fd, off, whence);
}
int32_t dg_ftell(DG_FILE *f) {
    if (!f || f->kind != 0) return -1;
    /* CupidOS vfs has no ftell — emulate via SEEK_CUR offset 0 if supported,
     * else track pos in DG_FILE (extend struct in v2). For now: */
    return vfs_seek(f->fd, 0, 1 /* SEEK_CUR */);
}
int dg_feof(DG_FILE *f)        { return f ? f->eof : 1; }
void dg_clearerr(DG_FILE *f)   { if (f) { f->eof = 0; f->err = 0; } }
int dg_fgetc(DG_FILE *f) {
    unsigned char c;
    if (dg_fread(&c, 1, 1, f) == 0) return -1;
    return c;
}
int dg_fputc(int c, DG_FILE *f) {
    unsigned char b = (unsigned char)c;
    return dg_fwrite(&b, 1, 1, f) == 1 ? c : -1;
}
char *dg_fgets(char *s, int n, DG_FILE *f) {
    int i = 0;
    while (i < n - 1) {
        int c = dg_fgetc(f);
        if (c < 0) { if (i == 0) return 0; break; }
        s[i++] = (char)c;
        if (c == '\n') break;

    s[i] = 0;
    return s;
}

/* ---- qsort (introsort, Sedgewick-style 3-way) ---- */
static void qs_swap(unsigned char *a, unsigned char *b, uint32_t sz) {
    while (sz--) { unsigned char t = *a; *a++ = *b; *b++ = t; }
}
void dg_qsort(void *base, uint32_t n, uint32_t sz,
              int (*cmp)(const void *, const void *)) {
    if (n < 2) return;
    unsigned char *a = (unsigned char *)base;
    /* simple median-of-three quicksort */
    uint32_t l = 0, r = n - 1;
    /* In a bounded-stack impl, recurse on the smaller partition.
     * For DOOM workload (lump table sort, ~3k entries) plain
     * recursive QS is fine. */
    /* Wrap in an iterative stack of (lo,hi) ranges. */
    typedef struct { uint32_t lo, hi; } range_t;
    range_t stk[64]; int top = 0;
    stk[top++] = (range_t){l, r};
    while (top > 0) {
        range_t cur = stk[--top];
        if (cur.lo >= cur.hi) continue;
        unsigned char *piv = a + ((cur.lo + cur.hi) / 2) * sz;
        unsigned char tmp[256];
        if (sz > 256) {
            /* fallback: simple insertion sort (rare path) */
            for (uint32_t i = cur.lo + 1; i <= cur.hi; i++)
                for (uint32_t j = i; j > cur.lo
                     && cmp(a + j*sz, a + (j-1)*sz) < 0; j--)
                    qs_swap(a + j*sz, a + (j-1)*sz, sz);
            continue;
        }
        memcpy(tmp, piv, sz);
        uint32_t i = cur.lo, j = cur.hi;
        while (1) {
            while (cmp(a + i*sz, tmp) < 0) i++;
            while (cmp(a + j*sz, tmp) > 0) j--;
            if (i >= j) break;
            qs_swap(a + i*sz, a + j*sz, sz);
            i++; if (j) j--;
        }
        if (j > cur.lo) stk[top++] = (range_t){cur.lo, j};
        stk[top++] = (range_t){j + 1, cur.hi};
    }
}

/* ---- setjmp/longjmp (x86-32) ----
 * Layout: [ebx esi edi ebp esp eip] = 6 dwords.
 * setjmp: save callee-saved + sp (after return-addr push) + return-addr.
 * longjmp: restore everything; jump to saved eip; eax = val (or 1 if val==0).
 */
__asm__ (
"   .global dg_setjmp\n"
"dg_setjmp:\n"
"   mov 4(%esp), %eax\n"        /* env */
"   mov %ebx,  0(%eax)\n"
"   mov %esi,  4(%eax)\n"
"   mov %edi,  8(%eax)\n"
"   mov %ebp, 12(%eax)\n"
"   lea 4(%esp), %ecx\n"
"   mov %ecx, 16(%eax)\n"        /* esp (caller's after ret pop) */
"   mov (%esp), %ecx\n"
"   mov %ecx, 20(%eax)\n"        /* eip */
"   xor %eax, %eax\n"
"   ret\n"

"   .global dg_longjmp\n"
"dg_longjmp:\n"
"   mov 4(%esp), %edx\n"          /* env */
"   mov 8(%esp), %eax\n"          /* val */
"   test %eax, %eax\n"
"   jne 1f\n"
"   mov $1, %eax\n"
"1: mov  0(%edx), %ebx\n"
"   mov  4(%edx), %esi\n"
"   mov  8(%edx), %edi\n"
"   mov 12(%edx), %ebp\n"
"   mov 16(%edx), %esp\n"
"   mov 20(%edx), %ecx\n"
"   jmp *%ecx\n"
);
```

```c
/* DOOM SFX lump format: 8-byte header (3, 0x00, samplerate, samples)
 * + N bytes u8 mono PCM. Convert at I_CacheSound time:
 * upsample 11025 → 22050 by 2x linear-interp, u8→s16 (subtract 128, scale).
 */

typedef struct {
    int16_t *pcm;           /* kmalloc'd */
    uint32_t frames;
    int      cached;
} sfx_cache_entry_t;

#define MAX_CACHED_SFX 256
static sfx_cache_entry_t s_cache[MAX_CACHED_SFX];

extern int W_GetNumForName(const char *name);
extern int W_LumpLength(int lump);
extern void *W_CacheLumpNum(int lump, int tag);

static int cache_sfx(sfxinfo_t *sfx) {
    if (sfx->lumpnum < 0 || sfx->lumpnum >= MAX_CACHED_SFX) return -1;
    sfx_cache_entry_t *e = &s_cache[sfx->lumpnum];
    if (e->cached) return 0;
    int len = W_LumpLength(sfx->lumpnum);
    uint8_t *raw = (uint8_t *)W_CacheLumpNum(sfx->lumpnum, 1);
    if (!raw || len < 8) return -1;
    /* parse header */
    uint32_t rate    = raw[2] | (raw[3] << 8);
    uint32_t samples = raw[4] | (raw[5] << 8) | (raw[6] << 16) | (raw[7] << 24);
    if (samples + 8 > (uint32_t)len) samples = (uint32_t)len - 8;
    /* allocate output: upsample by 22050/rate */
    uint32_t out_frames = (samples * 22050u) / (rate ? rate : 11025u);
    int16_t *out = (int16_t *)kmalloc(out_frames * 2);
    if (!out) return -1;
    /* linear-interp upsample */
    for (uint32_t i = 0; i < out_frames; i++) {
        uint32_t src_q16 = (i * (rate ? rate : 11025u) * 65536u) / 22050u;
        uint32_t si = src_q16 >> 16;
        uint32_t fr = src_q16 & 0xFFFF;
        if (si + 1 >= samples) si = samples - 2;
        int16_t s0 = (int16_t)((int)raw[8 + si] - 128) * 256;
        int16_t s1 = (int16_t)((int)raw[8 + si + 1] - 128) * 256;
        out[i] = (int16_t)(s0 + (((int32_t)(s1 - s0) * (int32_t)fr) >> 16));
    }
    e->pcm = out;
    e->frames = out_frames;
    e->cached = 1;
    return 0;
}

}

}

    uint8_t l = (uint8_t)((vol * (254 - sep)) / 254);
    uint8_t r = (uint8_t)((vol * sep) / 254);

}
static int cup_start_sound(sfxinfo_t *sfx, int channel, int vol, int sep, int pitch) {
```

Append to `i_sound_cupidos.c`:

```c
#include "../audio/midiopl.h"
#include "../audio/mus2midi.h"

extern int  W_GetNumForName(const char *name);
extern void *W_CacheLumpNum(int lump, int tag);
extern int  W_LumpLength(int lump);

static int s_music_inited = 0;
static uint8_t *s_midi_buf = 0;
static uint32_t s_midi_len = 0;
static uint32_t s_midi_pos = 0;
static int s_music_loop = 0;

static void music_pull(int16_t *out, uint32_t frames, void *ctx) {
    (void)ctx;
    /* feed bytes a fixed budget per fill — pace with PIT */
    /* naive: feed all remaining bytes the first time, render synth */
    if (s_midi_buf && s_midi_pos < s_midi_len) {
        /* feed up to 256 bytes per buffer fill */
        uint32_t chunk = s_midi_len - s_midi_pos;
        if (chunk > 256) chunk = 256;
        midiopl_feed(s_midi_buf + s_midi_pos, chunk);
        s_midi_pos += chunk;
        if (s_midi_pos >= s_midi_len && s_music_loop) {
            midiopl_reset();
            s_midi_pos = 0;
        }
    }
    midiopl_render(out, frames);
}

static int  cup_music_init(void) {
    int gn = W_GetNumForName("GENMIDI");
    if (gn < 0) {
        serial_write_string("[i_sound] no GENMIDI in WAD\n");
        return 0;
    }
    int len = W_LumpLength(gn);
    uint8_t *gm = (uint8_t *)W_CacheLumpNum(gn, 1);
    midiopl_init(gm, (uint32_t)len);
    s_music_inited = 1;
    return 1;
}
static void cup_music_shutdown(void) {}
static void cup_music_set_volume(int volume) {
    midiopl_set_volume((uint8_t)((volume * 127) / 15));
}
static void cup_music_pause(void)  { mixer_stop(8); }
static void cup_music_resume(void) {
    mixer_play_stream(8, music_pull, 0, 100, 100);
}
static void *cup_music_register(void *data, int len) {
    if (!s_music_inited) cup_music_init();
    if (s_midi_buf) { kfree(s_midi_buf); s_midi_buf = 0; }
    int rc = mus2midi_convert((const uint8_t *)data, (uint32_t)len,
                              &s_midi_buf, &s_midi_len);
    if (rc != 0) { serial_write_string("[i_sound] mus2midi failed\n"); return 0; }
    s_midi_pos = 0;
    return s_midi_buf;
}
static void cup_music_unregister(void *handle) {
    (void)handle;
    if (s_midi_buf) { kfree(s_midi_buf); s_midi_buf = 0; s_midi_len = 0; }
}
static void cup_music_play(void *handle, boolean looping) {
    (void)handle;
    s_music_loop = looping ? 1 : 0;
    s_midi_pos = 0;
    midiopl_reset();
    mixer_play_stream(8, music_pull, 0, 100, 100);
}
static void cup_music_stop(void) { mixer_stop(8); }
static boolean cup_music_is_playing(void) { return mixer_active(8); }
static void cup_music_poll(void) {}

music_module_t cupidos_music_module = {
    cup_devices, 1,
    cup_music_init,
    cup_music_shutdown,
    cup_music_set_volume,
    cup_music_pause,
    cup_music_resume,
    cup_music_register,
    cup_music_unregister,
    cup_music_play,
    cup_music_stop,
    cup_music_is_playing,
    cup_music_poll,
};
```

In `I_InitSound`, set `music_module = &cupidos_music_module;`.
